    Building b; b.x = x; b.y = y; b.type = type; b.faction = f;
    buildings.push_back(b);
}

void updateResources(Building& b) {
    if (b.type == 1) { // Quaestorium/шатер
        b.storage[FOOD] += 1; b.storage[MONEY] += 0.5; // пасив
    }

    SDL_Init(SDL_INIT_VIDEO);
    SDL_Window* win = SDL_CreateWindow("Punic Wars Mini-RTS", 1024, 768, 0, SDL_WINDOW_SHOWN);
    SDL_Renderer* ren = SDL_CreateRenderer(win, -1, SDL_RENDERER_ACCELERATED);
    
    // Рим табір

    addBuilding(200, 100, 1, ROME); // Квесторій
    
    // Карфаген
    addBuilding(800, 600, 0, CARTHAGE);
    
    bool running = true;
    Uint32 last_time = SDL_GetTicks();
    while (running) {
        SDL_Event e; while (SDL_PollEvent(&e)) if (e.type == SDL_QUIT) running = false;
        
        Uint32 now = SDL_GetTicks();
        if (now - last_time > 1000) { // 1сек update
            for (auto& b : buildings) updateResources(b);
            last_time = now;
        }
        
        SDL_SetRenderDrawColor(ren, 0x30, 0x50, 0x20, 0xFF); SDL_RenderClear(ren);
        // render buildings/units (додай rect/draw)
        SDL_RenderPresent(ren);
    }
    SDL_Quit();
    return 0;
}

cpp
#include <SDL2/SDL.h>
#include <string>
#include <vector>

enum Faction { ROME, CARTHAGE };
enum BuildingType { 
    HQ, STORAGE, BARRACKS_ROME, MARKET, // Рим
    HQ_CARTHAGE, TENT_LV1, TENT_LV2, TENT_LV3, STABLE, FOOD_STORAGE // Карфаген
};

struct Building {
    int x, y;
    BuildingType type;
    Faction faction;
    int level = 1; // для шатрів лівійців
    int storage_food = 0, storage_money = 0;
    int max_storage_food = 0, max_storage_money = 0;
    bool constructing = false;
    int construction_time = 0; // сек
    
    void init(BuildingType t, Faction f) {
        type = t; faction = f;
        switch(t) {
            case HQ: // Преторій
                if (f == ROME) { max_storage_food = 200; max_storage_money = 200; }
                break;
            case STORAGE: // Квесторій
                max_storage_food = 1000; max_storage_money = 1000;
                construction_time = 30;
                break;
            case BARRACKS_ROME: // Папіліо
                construction_time = 20;
                break;
            case MARKET: // Тенторіум
                construction_time = 15;
                break;
            case HQ_CARTHAGE: // Головне шатро
                max_storage_money = 1500;
                break;
            case TENT_LV1: // Шатер лівійців
                level = 1;
                construction_time = 25;
                break;
            case STABLE: // Загін коней
                construction_time = 35;
                break;
            case FOOD_STORAGE: // Сховище їжі Карфаген
                max_storage_food = 800;
                construction_time = 20;
                break;
        }
    }
    
    void upgrade() { // Покращення шатрів
        if (type == TENT_LV1 && level == 1) {
            level = 2; type = TENT_LV2; construction_time = 15;
        } else if (type == TENT_LV2 && level == 2) {
            level = 3; type = TENT_LV3; construction_time = 20;
        }
    }
    
    int getCapacity() { // Місткість шатра
        if (type == BARRACKS_ROME) return 8;
        if (type == TENT_LV1) return 6;
        if (type == TENT_LV2) return 8;
        if (type == TENT_LV3) return 10;
        if (type == STABLE) return 4;
        return 0;
    }
};

struct Unit {
    int x, y, hp, atk, defense;
    float speed;
    Faction faction;
    std::string unit_type; // "legionary", "phoenician", "peltast", "numidian"
    bool ranged = false;
    int range = 0;
    
    void init(std::string type, Faction f, int px, int py) {
        faction = f; unit_type = type; x = px; y = py;
        if (type == "legionary") { hp = 100; atk = 15; defense = 12; speed = 1.0f; }
        else if (type == "velites") { hp = 70; atk = 10; defense = 5; speed = 1.3f; ranged = true; range = 5; }
        else if (type == "phoenician") { hp = 90; atk = 14; defense = 10; speed = 1.0f; }
        else if (type == "peltast") { hp = 60; atk = 8; defense = 3; speed = 1.4f; ranged = true; range = 4; }
        else if (type == "numidian") { hp = 80; atk = 12; defense = 6; speed = 2.5f; }
        else if (type == "worker") { hp = 50; atk = 3; defense = 1; speed = 1.2f; }
    }
};

std::vector<Building> buildings;
std::vector<Unit> units;

void addBuilding(int x, int y, BuildingType type, Faction faction) {
    Building b;
    b.x = x; b.y = y;
    b.init(type, faction);
    buildings.push_back(b);
}

void spawnUnit(Building& barracks, std::string unit_type) {
    Unit u;
    u.init(unit_type, barracks.faction, barracks.x + 50, barracks.y + 50);
    units.push_back(u);
}

// morale_system.h
struct FactionState {
    bool has_food, has_money;
    float combat_modifier = 1.0f;
    
    void update(Faction f) {
        if (f == ROME) {
            if (!has_food) combat_modifier = 0.0f;   // втеча
            else if (!has_money) combat_modifier = 0.8f; // громадянський обов'язок
            else combat_modifier = 1.0f;
        }
        if (f == CARTHAGE) {
            if (!has_money) {
                // найманці повстають
                revolt = true;
                combat_modifier = 0.8f; // але воюють без їжі
            }
            else if (!has_food) combat_modifier = 0.8f;
            else combat_modifier = 1.0f;
        }
    }
    bool revolt = false; // Карфаген
    bool fleeing = false; // Рим
};
